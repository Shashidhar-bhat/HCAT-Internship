Simple Calculator with PyScript


from js import document def calculate(op): try: num1 = float(document.getElementById("num1").value) num2 = float(document.getElementById("num2").value) if op == '+': result = num1 + num2 elif op == '-': result = num1 - num2 elif op == '*': result = num1 * num2 elif op == '/': result = num1 / num2 if num2 != 0 else "Cannot divide by zero" else: result = "Invalid operator" document.getElementById("result").innerText = f"Result: {result}" except Exception as e: document.getElementById("result").innerText = f"Error: {e}" document.getElementById("add").addEventListener("click", lambda e: calculate('+')) document.getElementById("sub").addEventListener("click", lambda e: calculate('-')) document.getElementById("mul").addEventListener("click", lambda e: calculate('*')) document.getElementById("div").addEventListener("click", lambda e: calculate('/'))
